\begin{frame}
  \frametitle{\small Dereszynski et al. - Learning Probabilistic Behavior
    Models in RTS Games}

  \begin{block}{Problème de reconnaissance de stratégie}
    \begin{itemize}
    \item Stratégie = choix des unités et bâtiments produits.
    \item 1 observation toutes les 30s, pendant les 7 premières minutes
      de jeu.
    \end{itemize}
  \end{block}

  \pause
  
  \bigskip

  \begin{exampleblock}{Leur modèle}
    \begin{itemize}[<+->]
    \item Modèle appris sur 331 replays de joueurs pros.
    \item 30 états (ils ont fait plusieurs essais).
    \item Observations =  vecteur $v$ de booléens. $v[x]$  est vrai si
      au  moins  une unité  $x$  a  été  produite depuis  la  dernière
      observation.
    \item  Hypothèse que  les probabilités  de productions  des unités
      d'une même observation sont indépendantes (!!!).
    \end{itemize}
  \end{exampleblock}
  
\end{frame}
